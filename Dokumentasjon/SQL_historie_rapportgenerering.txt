//Prosjekt m. alle oppgaver

FROM Prosjekt p

LEFT JOIN Bruker ON Team.Teamleder = Bruker.ID

//m. alle milepæler
SELECT p.ID, p.Navn, p.Oppsummering, Team.Beskrivelse `Team`, CONCAT(Bruker.Fornavn, SPACE(1), Bruker.Mellomnavn, SPACE(1), Bruker.Etternavn) `Leder`, COUNT(Oppgave.Prosjekt_ID) `Oppgaver`, COUNT(Milepæl.Oppgave_ID) `Milepæler`
FROM Prosjekt p
LEFT JOIN KoblingTeamProsjekt ON p.ID = KoblingTeamProsjekt.Prosjekt_ID
LEFT JOIN Team ON KoblingTeamProsjekt.Team_ID = Team.ID
LEFT JOIN Bruker ON Team.Teamleder = Bruker.ID
LEFT JOIN Oppgave ON p.ID = Oppgave.Prosjekt_ID
LEFT JOIN Milepæl ON Oppgave.ID = Milepæl.Oppgave_ID
GROUP BY p.ID
